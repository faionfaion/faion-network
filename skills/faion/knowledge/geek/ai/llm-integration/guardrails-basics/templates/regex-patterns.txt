# purpose: Common regex patterns for input pre-filter (PII + token leakage).
# consumes: loaded by regex_layer in guardrail-stack.py
# produces: pattern list
# depends-on: content/01-core-rules.xml r1
# token-budget-impact: zero — runtime side

# Email addresses
[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,}

# SSN (US) 9 digits
\b\d{3}-\d{2}-\d{4}\b

# Credit card-ish
\b(?:\d[ -]*?){13,16}\b

# Bearer tokens / API keys
(?i)(sk-|ghp_|gho_|github_pat_|AIza|AKIA)[A-Za-z0-9_-]{20,}

# Common prompt-injection markers
(?i)ignore (all )?previous (instructions|rules)
(?i)you are (now )?DAN
(?i)forget (all )?(previous|prior) (instructions|guidelines)
