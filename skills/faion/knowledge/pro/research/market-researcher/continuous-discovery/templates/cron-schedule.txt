# Continuous discovery cron schedule
# Requires: CLAUDE_API_KEY set; claude CLI in PATH
# Outputs to .aidocs/market-intel/

# Quick market pulse (every 30 min, 6am-10pm)
*/30 6-22 * * *  /usr/local/bin/claude run /market-pulse-quick

# Daily market digest
0 7 * * *        /usr/local/bin/claude run /market-daily

# Weekly synthesis (Monday morning)
0 8 * * 1        /usr/local/bin/claude run /market-weekly

# Bi-weekly synthesizer (1st and 15th)
0 9 1,15 * *     /usr/local/bin/claude run /market-synth

# Monthly strategist (1st of month)
0 10 1 * *       /usr/local/bin/claude run /market-strategist

# Quarterly TAM refresh (1st of Jan/Apr/Jul/Oct)
0 11 1 1,4,7,10 * /usr/local/bin/claude run /tam-refresh
